    let TOKEN = localStorage.getItem("ce_token") || null;
    let ME = null, socket = null, jailInterval = null;
    async function req(path, opts={}) {
    const h = {"Content-Type":"application/json"};
    if (TOKEN) h["Authorization"] = `Bearer ${TOKEN}`;
    const r = await fetch(path, {...opts, headers:h});
    const d = await r.json();
    if (!r.ok) throw new Error(d.error || "خطأ");
    return d;
    }
    let toastT;
    function toast(msg, type="inf") {
    const t = document.getElementById("toast");
    t.textContent = msg; t.className = `on ${type}`;
    clearTimeout(toastT); toastT = setTimeout(()=>t.className="",3000);
    }
    function log(msg, type="inf") {
    const box = document.getElementById("logBox");
    const el = document.createElement("div");
    el.className = `ll ${type}`;
    const t = new Date().toLocaleTimeString("ar",{hour:"2-digit",minute:"2-digit"});
    el.innerHTML = `${msg} <span class="lt">${t}</span>`;
    box.insertBefore(el, box.firstChild);
    while(box.children.length > 80) box.lastChild.remove();
    }
    function startJailTimer(jailUntil) {
    clearInterval(jailInterval);
    const banner = document.getElementById("jailBanner");
    const span = document.getElementById("jailTimer");
    function tick() {
    const left = jailUntil - Date.now();
    if (left <= 0) { banner.classList.remove("on"); clearInterval(jailInterval); toast("خرجت من السجن! 🔓","ok"); loadMe(); return; }
    const m = Math.floor(left/60000), s = Math.floor((left%60000)/1000);
    span.textContent = `متبقي: ${m}:${s.toString().padStart(2,"0")}`;
    }
    banner.classList.add("on"); tick(); jailInterval = setInterval(tick,1000);
    }
    function updateStats(p) {
    document.getElementById("sMoney").textContent = Number(p.money).toLocaleString("ar");
    document.getElementById("sHeat").textContent = p.heat;
    document.getElementById("sStr").textContent = p.strength;
    document.getElementById("sSte").textContent = p.stealth;
    document.getElementById("sInt").textContent = p.intelligence;
    document.getElementById("bHeat").style.width = `${Math.min(100,p.heat)}%`;
    document.getElementById("hRole").textContent = p.role.toUpperCase();
    if (p.role==="admin") document.getElementById("adminTab").style.display="";
    if (p.jail_until > Date.now()) startJailTimer(p.jail_until);
    else document.getElementById("jailBanner").classList.remove("on");
    }
    function switchTab(tab, btn) {
    document.querySelectorAll(".tab").forEach(t=>t.classList.remove("on")); btn.classList.add("on");
    document.getElementById("fLogin").style.display = tab==="login"?"block":"none";
    document.getElementById("fReg").style.display = tab==="reg"?"block":"none";
    }
    async function login() {
    const username = document.getElementById("lUser").value.trim();
    const password = document.getElementById("lPass").value;
    document.getElementById("lErr").textContent = "";
    if (!username||!password) return document.getElementById("lErr").textContent="أدخل الاسم وكلمة المرور";
    const btn = document.getElementById("lBtn"); btn.disabled=true; btn.textContent="جارٍ الدخول...";
    try {
    const data = await req("/login",{method:"POST",body:JSON.stringify({username,password})});
    TOKEN = data.token; localStorage.setItem("ce_token",TOKEN); await enterGame();
    } catch(e) { document.getElementById("lErr").textContent=e.message; }
    finally { btn.disabled=false; btn.textContent="دخول"; }
    }
    async function register() {
    const username = document.getElementById("rUser").value.trim();
    const password = document.getElementById("rPass").value;
    document.getElementById("rErr").textContent = "";
    if (!username||!password) return document.getElementById("rErr").textContent="أدخل الاسم وكلمة المرور";
    const btn = document.getElementById("rBtn"); btn.disabled=true; btn.textContent="جارٍ التسجيل...";
    try {
    await req("/register",{method:"POST",body:JSON.stringify({username,password})});
    toast("تم إنشاء الحساب! سجّل دخولك","ok");
    document.querySelector(".tab").click();
    document.getElementById("lUser").value = username;
    } catch(e) { document.getElementById("rErr").textContent=e.message; }
    finally { btn.disabled=false; btn.textContent="إنشاء حساب"; }
    }
    async function loadMe() { const d=await req("/me"); ME=d; updateStats(d); return d; }
    async function enterGame() {
    ME = await loadMe();
    document.getElementById("hUser").textContent = ME.username;
    document.getElementById("sLogin").classList.remove("on");
    document.getElementById("sGame").classList.add("on");
    log(`مرحباً ${ME.username}!`,"gd");
    connectSocket();
    }
    function logout() {
    TOKEN=null; ME=null; localStorage.removeItem("ce_token");
    if(socket) socket.disconnect(); clearInterval(jailInterval);
    document.getElementById("sGame").classList.remove("on");
    document.getElementById("sLogin").classList.add("on");
    }
    function connectSocket() {
    socket = io();
    socket.emit("join", TOKEN);
    socket.on("attacked",({by,lost})=>{
    toast(`⚠️ ${by} هاجمك وسرق ${Number(lost).toLocaleString()} $!`,"err");
    log(`💥 ${by} هاجمك وسرق ${Number(lost).toLocaleString()} $`,"fl");
    loadMe();
    });
    socket.on("playerOnline",({username})=>log(`🟢 ${username} اتصل`,"inf"));
    socket.on("playerOffline",({username})=>log(`🔴 ${username} انقطع`,"inf"));
    }
    function nav(page, btn) {
    document.querySelectorAll(".pg").forEach(p=>p.classList.remove("on"));
    document.querySelectorAll(".nb").forEach(b=>b.classList.remove("on"));
    document.getElementById(`pg-${page}`).classList.add("on"); btn.classList.add("on");
    if(page==="missions") loadMissions();
    if(page==="attack") loadPlayers();
    if(page==="skills") renderSkills();
    if(page==="lb") loadLeaderboard();
    if(page==="admin") loadAdmin();
    }
    async function loadMissions() {
    try {
    const missions = await req("/missions");
    document.getElementById("missionList").innerHTML = missions.map(m=>{
    const locked=(m.minStealth&&ME.stealth<m.minStealth)||(m.minStr&&ME.strength<m.minStr)||(m.minIntel&&ME.intelligence<m.minIntel);
    const req2=m.minStealth?`تخفي ${m.minStealth}`:m.minStr?`قوة ${m.minStr}`:m.minIntel?`ذكاء ${m.minIntel}`:"متاحة";
    return `<div class="mi ${locked?'lk':'av'}">
    <div><div class="mn">${locked?'🔒 ':''}${m.name}</div>
    <div class="mm">🔥 +${m.heatGain} &nbsp;|&nbsp; متطلب: ${req2}</div>
    ${!locked?`<button class="btn-m" onclick="doMission(${m.index})">تنفيذ</button>`:''}</div>
    <div class="mr">${Number(m.reward[0]).toLocaleString()}–${Number(m.reward[1]).toLocaleString()}<span>دولار</span></div></div>`;
    }).join("")||`<div class="empty">لا توجد مهام</div>`;
    } catch(e) { toast(e.message,"err"); }
    }
    async function doMission(index) {
    try {
    const data = await req("/mission",{method:"POST",body:JSON.stringify({missionIndex:index})});
    await loadMe();
    if(data.success) { log(`✅ ${data.mission} — ربحت ${Number(data.reward).toLocaleString()} $`,"ok"); toast(`+${Number(data.reward).toLocaleString()} $`,"ok"); }
    else if(data.jailed) { log(`🚔 القي عليك القبض!`,"fl"); toast("أنت في السجن 🔒","err"); startJailTimer(data.jail_until); }
    loadMissions();
    } catch(e) { toast(e.message,"err"); }
    }
    async function loadPlayers() {
    try {
    const players = await req("/leaderboard");
    document.getElementById("playerList").innerHTML = players.filter(p=>p.username!==ME.username).map(p=>`
    <div class="pi"><div><div class="pn">${p.username}</div>
    <div class="pm">💰 ${Number(p.money).toLocaleString()} $ &nbsp;|&nbsp; 💪 ${p.strength}</div></div>
    <button class="btn-atk" onclick="attack('${p.id}')">⚔ هجوم</button></div>`
    ).join("")||`<div class="empty">لا يوجد لاعبون آخرون</div>`;
    } catch(e) { toast(e.message,"err"); }
    }
    async function attack(targetId) {
    try {
    const data = await req("/attack",{method:"POST",body:JSON.stringify({targetId})});
    await loadMe();
    if(data.success) { log(`⚔️ هجوم ناجح — سرقت ${Number(data.stolen).toLocaleString()} $`,"ok"); toast(`+${Number(data.stolen).toLocaleString()} $`,"ok"); }
    else { log(`💥 فشل الهجوم`,"fl"); toast("فشل الهجوم","err"); }
    loadPlayers();
    } catch(e) { toast(e.message,"err"); }
    }
    function renderSkills() {
    if(!ME) return;
    const skills=[{key:"strength",icon:"💪",name:"القوة",val:ME.strength},{key:"stealth",icon:"🥷",name:"التخفي",val:ME.stealth},{key:"intelligence",icon:"🧠",name:"الذكاء",val:ME.intelligence}];
    document.getElementById("skillCards").innerHTML = skills.map(s=>{
    const cost=s.val*5000, can=ME.money>=cost;
    return `<div class="ski"><div class="ski-ico">${s.icon}</div><div class="ski-n">${s.name}</div>
    <div class="ski-l">${s.val}</div><div class="ski-c">التكلفة: ${Number(cost).toLocaleString()} $</div>
    <button class="btn-up" onclick="upgrade('${s.key}')" ${can?'':'disabled'}>${can?'ترقية ↑':'مال غير كافٍ'}</button></div>`;
    }).join("");
    }

    try {
    const data = await req("/upgrade",{method:"POST",body:JSON.stringify({stat})});
    await loadMe(); log(`📈 رقّيت ${stat} إلى مستوى ${data.newLevel}`,"gd"); toast(`ترقية ناجحة! مستوى ${data.newLevel}`,"ok"); renderSkills();
    } catch(e) { toast(e.message,"err"); }
    }

    const data = await req("/surrender",{method:"POST"});

    } catch(e) { toast(e.message,"err"); }
    }

    const code = prompt("أدخل كود الأدمن:");

    catch(e) { toast(e.message,"err"); }
    }

    try { await req(`/admin/release/${id}`,{method:"POST"}); toast("تم الإفراج","ok"); loadAdmin(); }

    }
